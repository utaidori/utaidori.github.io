AutoHotKey v2 で Chrome のマウスジェスチャーを実現する

  • 概要：アクティブウインドウが Chrome のとき有効になるマウスジェスチャー
  • 機能
    • 右クリックを押下中にマウスホイールで拡大 / 縮小（ズーム）
    • 左クリックを押下中に右クリック押下で全画面
    • 右クリックを押下中にマウスジェスチャーで各種操作（操作の詳細は下記コード内参照）
  • 挙動：各マウス操作に対してキーを入力

AutoHotKey v2

#Requires AutoHotkey v2.0
#MaxThreadsPerHotkey 2

#HotIf WinActive("ahk_exe chrome.exe")
; ズーム操作が行われたかを判定するフラグ
global zoomUsed := false

; --- ズーム操作（右クリックしながらホイール） ---
~RButton & WheelUp:: {
    global zoomUsed := true
    Send("^{+}")
}

~RButton & WheelDown:: {
    global zoomUsed := true
    Send("^-")
}

; --- メインのジェスチャー処理 ---
RButton:: {
    global zoomUsed
    zoomUsed := false ; フラグのリセット

    ; 1. LButton + RButton → F11 (同時押し判定)
    if GetKeyState("LButton", "P") {
        Send("{F11}")
        return
    }

    ; 座標と履歴の初期化
    MouseGetPos(&sx, &sy)
    gesture := ""
    lastDir := ""
    threshold := 60 ; 判定のしきい値（ピクセル）

    ; タイムアウト設定（3秒以上押しっぱなしなら強制終了して右クリックを出す）
    startTime := A_TickCount
    
    ; 右ボタンが離されるまでループ
    while GetKeyState("RButton", "P") {
        ; ズームが使われたら即座にループを抜ける（ジェスチャーを無効化）
        if (zoomUsed || A_TickCount - startTime > 3000)
            break

        MouseGetPos(&cx, &cy)
        dx := cx - sx
        dy := cy - sy
        dist := Sqrt(dx**2 + dy**2)

        if (dist > threshold) {
            ; 方向判定
            dir := ""
            if (Abs(dx) > Abs(dy)) {
                dir := (dx > 0) ? "R" : "L"
            } else {
                dir := (dy > 0) ? "D" : "U"
            }

            ; 新しい方向であれば履歴に追加し、基準点を更新
            if (dir != lastDir) {
                gesture .= dir
                lastDir := dir
                sx := cx
                sy := cy
            }
        }
        Sleep(10)
    }

    ; 2. ボタンを離した後のアクション分岐
    if (zoomUsed) {
        ; ズーム使用後は何もしない
    } else if (gesture == "") {
        ; 移動がなければ通常の右クリックを送信
        Send("{Blind}{RButton}")
    } else {
        ; ジェスチャー履歴に基づいて実行
        switch gesture {
            ; --- 1アクション ---
            case "L": Send("{XButton1}")  ; 戻る
            case "R": Send("{XButton2}")  ; 進む
            case "U": Send("{Home}")      ; ページ先頭
            case "D": Send("^w")          ; タブを閉じる

            ; --- 2アクション ---
            case "LR": Send("^+t")        ; 左→右：閉じたタブを開く
            case "RL": Send("^^t")        ; 右→左：新しいタブを開く
            case "LU": Send("{HOME}")     ; 左→上：ページ先頭
            case "RD": Send("{END}")      ; 右→下：ページ末尾
            case "UD": Send("{F5}")       ; 上→下：更新

            ; 定義されていない動きはツールチップで表示（デバッグ用）
            default:
                ToolTip("Gesture: " . gesture)
                SetTimer(() => ToolTip(), -1000)
        }
    }
}
#HotIf
  • 設定の変更
    • threshold の値（ピクセル）：マウスジェスチャーの判定のしきい値
      （右クリック押下後、方向転換または離すまでにマウスカーソルを移動させた x, y の値がこの値を超えるとマウスジェスチャーとして認める）
    • switch ブロックの case "xx" (LRUD）：反応するマウスジェスチャー
    • switch ブロックの Send のキー：マウスジェスチャーに対応して入力されるキー
